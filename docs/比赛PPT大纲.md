```
┌──────────────┐     ┌──────────────┐           │
│  实时检测    │ ←── │  Deno沙箱    │ ←─────────┘
│ 发现漏洞     │     │ 安全执行     │  代码验证+修复
└──────────────┘     └──────────────┘  (最多3次尝试)
```

**代码示例**（AI生成的插件片段）：
```typescript
// AI根据"MySQL + 数字型参数"特征生成
export function scan_response(ctx: CombinedContext): void {
  const userId = new URL(ctx.request.url).searchParams.get('user_id');
  
  // 针对MySQL的错误特征检测
  const mysqlErrors = [
    /You have an error in your SQL syntax/i,
    /MySQLSyntaxErrorException/i
  ];
  
  if (userId && /^\d+$/.test(userId)) {
    for (const pattern of mysqlErrors) {
      if (pattern.test(ctx.response.body)) {
        emit_finding({ vuln_type: "sqli", severity: "critical" });
      }
    }
  }
}
```
